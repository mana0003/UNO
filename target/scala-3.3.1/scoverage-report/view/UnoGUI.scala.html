40 class GameState(gui: UnoGUI, controller: UnoController) extends State {
41   override def display(pane: Pane): Unit = {
42     pane.children.clear()
43 
44     val playerLabel = new scalafx.scene.control.Label(s"Current player: Player ${controller.field.currentPlayer + 1}")
45     val topCardLabel = new scalafx.scene.control.Label(s"Current top card: ${controller.field.topCard.value}") {
46       textFill = controller.field.topCard.getColorCode
47     }
48 
49     val handListView = new ListView[Card] {
50       items = scalafx.collections.ObservableBuffer(controller.field.players(controller.field.currentPlayer).hand.cards: _*)
51 
52       cellFactory = new Callback[ListView[Card], ListCell[Card]] {
53         override def call(param: ListView[Card]): ListCell[Card] = {
54           new ListCell[Card] {
55             item.onChange { (_, _, cardOpt) =>
56               val cardOption = Option(cardOpt)
57               text = cardOption.map(_.value.toString).getOrElse("")
58               textFill = cardOption.map(_.getColorCode).getOrElse(Color.Pink)
59             }
60           }
61         }
62       }
63     }
64 
65     val drawButton = new scalafx.scene.control.Button("Draw Card") {
66       onAction = _ => {
67         controller.draw()
68         Platform.runLater(() => {
69           gui.display()
70         })
71       }
72     }
73 
74     val playButton = new scalafx.scene.control.Button("Play Card") {
75       onAction = _ => {
76         val selectedCardIndex = handListView.selectionModel().getSelectedIndex
77         if (selectedCardIndex >= 0) {
78           val card = controller.field.players(controller.field.currentPlayer).hand.cards(selectedCardIndex)
79           if (controller.field.players(controller.field.currentPlayer).valid(card) && card.canBePlayedOn(controller.field.topCard)) {
80             controller.play(card)
81             Platform.runLater(() => {
82               gui.display()
83             })
84           } else {
85             showAlert("Invalid Move", "The selected card cannot be played.")
86           }
87         } else {
88           showAlert("No Card Selected", "Please select a card to play.")
89         }
90       }
91     }
92 
93     val undoButton = new scalafx.scene.control.Button("Undo") {
94       onAction = _ => {
95         controller.undo()
96         Platform.runLater(() => gui.display())
97       }
98     }
99 
100     val redoButton = new scalafx.scene.control.Button("Redo") {
101       onAction = _ => {
102         controller.redo()
103         Platform.runLater(() => gui.display())
104       }
105     }
106 
107     val quitButton = new scalafx.scene.control.Button("Quit") {
108       onAction = _ => {
109         controller.notifyObservers(Event.Quit)
110         Platform.exit()
111       }
112     }
113 
114     val buttonLayout = new scalafx.scene.layout.HBox {
115       spacing = 10
116       children = Seq(drawButton, playButton, undoButton, redoButton, quitButton)
117     }
118 
119     val layout = new scalafx.scene.layout.VBox {
120       spacing = 20
121       children = Seq(playerLabel, topCardLabel, handListView, buttonLayout)
122     }
123     pane.children.add(layout)
124   }
125 
126   private def showAlert(alertTitle: String, message: String): Unit = {

128       initOwner(gui.stage)
129       this.title = alertTitle
130       headerText = None
131       contentText = message
132     }.showAndWait()
133   }
134 }

136 class UnoGUI(controller: UnoController) extends JFXApp3 with Observer {
137   controller.addObserver(this)
138 
139   private var state: State = new BeginState(this, controller)
140   private val menuPane: Pane = new Pane
141   private val gamePane: Pane = new Pane
142   private val rootPane: StackPane = new StackPane {
143     children = List(menuPane, gamePane)
144   }
145 
146   def setState(newState: State): Unit = {
147     state = newState
148   }
149 
150   def display(): Unit = {
151     state.display(gamePane)
152     gamePane.toFront()
153   }
154 
155   override def start(): Unit = {
156     controller.setGuiActive(true) // new
157     stage = new PrimaryStage {
158       title = "Uno Game"
159       width = 800
160       height = 600
161       scene = new Scene {
162         fill = Color.Pink
163         content = rootPane
164       }
165     }
166     stage.scene().onKeyPressed = handleKeyPress
167     display()
168   }
169 
170   private def handleKeyPress: KeyEvent => Unit = {
171     case ke: KeyEvent if ke.code == KeyCode.Left =>
172       state = new BeginState(this, controller)
173       display()
174     case _ =>
175   }
176 
177   override def update(e: Event): Unit = {
178     println(s"GUI Received event: $e")
179     e match {
180       case Event.Start =>
181         Platform.runLater(() => {
182           state = new GameState(this, controller)
183           display()
184         })
185       case Event.Play | Event.Draw | Event.Undo | Event.Redo | Event.Error =>
186         Platform.runLater(() => {
187           state.display(gamePane) // Refresh the game state
188         })
189       case Event.Quit =>
190         Platform.exit()
191     }
192   }
193 }
