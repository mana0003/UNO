11   private def processInputLine(input: Int, handSize: Int): Unit = {
12     if (input < 1 || input > handSize) {
13       println("Invalid card number. Please try again.")
14       gameContinue()
15     } else {
16       val card = controller.getField.players(controller.getCurrentPlayer).hand.cards(input - 1)
17       if (controller.getField.players(controller.getCurrentPlayer).valid(card) && card.canBePlayedOn(controller.getField.topCard)) {
18         UnoActionBuilder.builder().setAction("play").setCard(card).build().executeAction(controller, controller.getField.players(controller.getCurrentPlayer))
19       } else {
20         println("Card does not fit. Do you want to:")
21         println("1. Draw a card")
22         println("2. Try again")
23         println("3. Undo")
24         println("4. Redo")
25         println("5. Quit")
26         val readLine = StdIn.readLine().toIntOption.getOrElse(-1)
27         readLine match {
28           case 1 => UnoActionBuilder.builder().setAction("draw").build().executeAction(controller, controller.getField.players(controller.getCurrentPlayer))
29           case 2 => gameContinue()
30           case 3 => undoAction()
31           case 4 => redoAction()
32           case 5 => controller.notifyObservers(Event.Quit)
33           case _ =>
34             println("Invalid input. Please enter a number.")
35             gameContinue()
36         }
37       }
38     }
39   }
40 
41   override def update(e: Event): Unit = {
42     if (controller.isGuiMode) {
43       return
44     }

60         gameContinue()

62         println("An error occurred!")
63         gameContinue()
64       case Event.Quit =>
65         gameOver()
66     }
67   }

69   def startGame(): Unit = {

117     if (cardNumber.isEmpty) {

119       gameContinue()
