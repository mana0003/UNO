14   private def processInputLine(input: Int, handSize: Int): Unit = {
15     if (input < 1 || input > handSize) {
16       println("Invalid card number. Please try again.")
17       gameContinue()
18     } else {
19       val card = controller.field.players(controller.field.currentPlayer).hand.cards(input - 1)
20       if (controller.field.players(controller.field.currentPlayer).valid(card) && card.canBePlayedOn(controller.field.topCard)) {
21         controller.play(card)
22       } else {
23         println("Card does not fit. Do you want to:")
24         println("1. Draw a card")
25         println("2. Try again")
26         println("3. Quit")
27         val readLine = StdIn.readLine().toIntOption.getOrElse(-1)
28         readLine match {
29           case 1 => controller.draw()
30           case 2 => gameContinue()
31           case 3 => controller.notifyObservers(Event.Quit)
32           case _ =>
33             println("Invalid input. Please enter a number.")
34             gameContinue()
35         }
36       }
37     }
38   }

40   override def update(e: Event): Unit = {

50   }
